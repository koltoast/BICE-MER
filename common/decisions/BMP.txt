				ideology = neutrality
				popularity = -0.05
				}

			clr_country_flag = BMP_Riots_WestBank

				limit = {
					NOT = { has_country_flag = BMP_Riots_Gaza }

				}
				remove_ideas = BMP_Riots
				clr_country_flag = crisis_in_progress
			}
